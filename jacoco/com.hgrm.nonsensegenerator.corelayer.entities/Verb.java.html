package com.hgrm.nonsensegenerator.corelayer.entities;

import com.fasterxml.jackson.annotation.JsonProperty;
import com.hgrm.nonsensegenerator.corelayer.enums.Tense;
import com.hgrm.nonsensegenerator.corelayer.enums.WordType;
import lombok.Getter;

@Getter
public class Verb extends Word {
    Tense tense;

    public Verb(@JsonProperty("value") String value, @JsonProperty("tense") Tense tense) {
        this.value = value;
        this.tense = tense;
        this.type = WordType.VERB;
    }

    @Override
    public boolean equals(Object o) {
        if (o == null)
            return false;
        if (o.getClass() == this.getClass()) {
            Verb word = (Verb) o;
            return this.value.equals(word.value) && this.tense.equals(word.tense) && this.type == word.type;
        }
        return false;
    }
}
