package com.hgrm.nonsensegenerator.corelayer.factories;

import com.hgrm.nonsensegenerator.corelayer.entities.Adjective;
import com.hgrm.nonsensegenerator.corelayer.entities.Noun;
import com.hgrm.nonsensegenerator.corelayer.entities.Verb;
import com.hgrm.nonsensegenerator.corelayer.entities.Word;
import com.hgrm.nonsensegenerator.corelayer.enums.Tense;
import com.hgrm.nonsensegenerator.corelayer.enums.WordType;
import org.springframework.stereotype.Component;

@Component
public class WordFactory {

    public WordFactory() {
        //Only for Spring IoC
    }

    public Word createWord(String wordString, WordType wordType) {
        return switch (wordType) {
            case NOUN -> new Noun(wordString);
            case VERB -> throw new IllegalArgumentException("Verb creation requires tense");
            case ADJECTIVE -> new Adjective(wordString);
        };
    }
    public Word createWord(String wordString, WordType wordType, Tense tense) {
        return switch (wordType) {
            case NOUN -> new Noun(wordString);
            case VERB -> new Verb(wordString, tense);
            case ADJECTIVE -> new Adjective(wordString);
        };
    }
}
