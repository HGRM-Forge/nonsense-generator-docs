package com.hgrm.nonsensegenerator.corelayer.enums;

public enum Tense {
    PAST() {
        @Override
        public String toString() {
            return "Past";
        }
    },
    PRESENT() {
        @Override
        public String toString() {
            return "Present";
        }
    },
    FUTURE() {
        @Override
        public String toString() {
            return "Future";
        }
    };

    @Override
    public abstract String toString();
    public static Tense fromString(String string) {
        return switch (string.toLowerCase()) {
            case "past" -> PAST;
            case "present" -> PRESENT;
            case "future" -> FUTURE;
            default -> throw new IllegalArgumentException("Invalid tense string: " + string);
        };
    }
}
