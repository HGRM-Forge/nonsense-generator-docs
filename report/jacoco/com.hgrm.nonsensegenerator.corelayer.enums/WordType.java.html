package com.hgrm.nonsensegenerator.corelayer.enums;

public enum WordType {
    NOUN() {
        @Override
        public String toString() {
            return "Noun";
        }
    },
    VERB() {
        @Override
        public String toString() {
            return "Verb";
        }
    },
    ADJECTIVE() {
        @Override
        public String toString() {
            return "Adjective";
        }
    };
    @Override
    public abstract String toString();
    public static WordType fromString(String word) {
        return switch (word.toLowerCase()) {
            case "noun" -> WordType.NOUN;
            case "verb" -> WordType.VERB;
            case "adjective" -> WordType.ADJECTIVE;
            default -> throw new IllegalArgumentException("Unknown word type: " + word);
        };
    }
}
