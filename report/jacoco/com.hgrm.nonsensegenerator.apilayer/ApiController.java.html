package com.hgrm.nonsensegenerator.apilayer;

import com.hgrm.nonsensegenerator.corelayer.CoreController;
import com.hgrm.nonsensegenerator.corelayer.entities.SentenceProcessingOption;
import com.hgrm.nonsensegenerator.corelayer.entities.SentenceTemplate;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.web.bind.annotation.*;

import java.util.List;

@RestController
@CrossOrigin(origins = "*")
public class ApiController {
    @Autowired
    CoreController coreController;

    @GetMapping("/api/get-sentence-history")
    public List<String> getSentenceHistory() {
        return coreController.getSentenceHistory();
    }

    @PostMapping("/api/process-and-generate-sentence")
    public CoreController.SentenceOutputData processAndGenerateSentence(@RequestBody SentenceInputData inputData) {
        return coreController.processAndGenerateSentence(inputData.sentence(), inputData.option());
    }

    @GetMapping("/api/get-template-list")
    public List<SentenceTemplate> getTemplateList() {
        return coreController.getTemplateList();
    }

    public record SentenceInputData(String sentence, SentenceProcessingOption option ) {}
}
