11  @NoArgsConstructor
12  @AllArgsConstructor
13  public class SentenceTemplate {
14      @Getter
15      @NonNull
16      private String templatePlaceholder;
17  
18      private static final String PLACEHOLDER_PREFIX = "%%";
19  
20      public int getWordCount(WordType wordType) {
21          String placeholder = switch (wordType) {
22              case NOUN -> PLACEHOLDER_PREFIX + "noun";
23              case VERB -> PLACEHOLDER_PREFIX + "verb";
24              case ADJECTIVE -> PLACEHOLDER_PREFIX + "adjective";
25          };
26          return (templatePlaceholder.length() - templatePlaceholder.replace(placeholder, "").length()) / placeholder.length();
27      }
28  
29      public String buildSentence(List<Word> words) {
30          String sentence = templatePlaceholder;
31  
32          for (Word word : words) {
33              String placeholder = switch (word.getType()) {
34                  case NOUN -> PLACEHOLDER_PREFIX +"noun";
35                  case VERB -> PLACEHOLDER_PREFIX +"verb";
36                  case ADJECTIVE -> PLACEHOLDER_PREFIX +"adjective";
37              };
38              sentence = sentence.replaceFirst(placeholder, word.getValue());
39          }
40  
41          if (sentence.contains(PLACEHOLDER_PREFIX)) {
42              throw new IllegalArgumentException("Not enough words provided for the template: "+sentence);
43          }
44          return sentence;
45      }
