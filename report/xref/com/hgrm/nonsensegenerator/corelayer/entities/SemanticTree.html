10  import lombok.Getter;
11  import lombok.ToString;
12  
13  import javax.annotation.Nullable;
14  
15  @Getter @ToString
16  public class SemanticTree {
17      private Tense tense;
18      private final List<Word> nounList = new ArrayList<>();
19      private final List<Word> verbList = new ArrayList<>();
20      private final List<Word> adjectiveList = new ArrayList<>();
21      private final Random random = new Random();
22      private final List<SemanticTreeEdge> rootEdgeList = new ArrayList<>();
23  
24      public void addRootEdge(SemanticTreeEdge rootEdge) {
25          rootEdgeList.add(rootEdge);
26      }
27  
28      public void addWord(Word word) {
29          switch (word.getType()) {
30  
31              case NOUN:
32                  nounList.add(word);
33                  break;
34              case VERB:
35                  verbList.add(word);
36                  if (tense == null) {
37                      Verb verb = (Verb) word;
38                      tense = verb.getTense();
39                  }
40                  break;
41              case ADJECTIVE:
42                  adjectiveList.add(word);
43          }
44      }
45  
46      public List<Word> getWordList() {
47          List<Word> allWords = new ArrayList<>(nounList.size() + verbList.size() + adjectiveList.size());
48          allWords.addAll(nounList);
49          allWords.addAll(verbList);
50          allWords.addAll(adjectiveList);
51          return allWords;
52      }
53  
54      public Word getRandomWord(WordType wordType) {
55          return getRandomWord(wordType, this.tense);
56      }
57      public Word getRandomWord(WordType wordType, @Nullable Tense tense) {
58          return switch (wordType) {
59              case NOUN -> {
60                  if (nounList.isEmpty())
61                      yield null;
62                  yield nounList.get(random.nextInt(nounList.size()));
63              }
64              case VERB -> {
65                  if (verbList.isEmpty())
66                      yield null;
67                  yield getRandomVerb(tense);
68              }
69              case ADJECTIVE -> {
70                  if (adjectiveList.isEmpty())
71                      yield null;
72                  yield adjectiveList.get(random.nextInt(adjectiveList.size()));
73              }
74          };
75      }
76      private Word getRandomVerb(Tense tense) {
77          List<Word> verbsWithTense = getVerbList(tense);
78          if (verbsWithTense.isEmpty()) {
79              return null;
80          }
81          return verbsWithTense.get(random.nextInt(verbsWithTense.size()));
82      }
83      public List<Word> getVerbList(Tense tense) {
84          ArrayList<Word> verbsWithTense = new ArrayList<>();
85          for (Word word : verbList) {
86              Verb verb = (Verb) word;
87              if ((verb.getTense() == tense) && (!verbsWithTense.contains(verb))) {
88                  verbsWithTense.add(verb);
89              }
90          }
91          return verbsWithTense;
92      }
93  }
