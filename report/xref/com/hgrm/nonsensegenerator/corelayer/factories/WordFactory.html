18      public Word createWord(String wordString, WordType wordType) {
19          return switch (wordType) {
20              case NOUN -> new Noun(wordString);
21              case VERB -> throw new IllegalArgumentException("Verb creation requires tense");
22              case ADJECTIVE -> new Adjective(wordString);
23          };
24      }
25      public Word createWord(String wordString, WordType wordType, Tense tense) {
26          return switch (wordType) {
27              case NOUN -> new Noun(wordString);
28              case VERB -> new Verb(wordString, tense);
29              case ADJECTIVE -> new Adjective(wordString);
30          };
31      }
