13  @SpringBootTest
14  class WordFactoryTest {
15      @Autowired
16      WordFactory wordFactory;
17  
18      // Assert that the createWord() method works correctly for different word types and different tenses
19      @Test
20      void testCreateWord() {
21          String wordString = "test";
22  
23          // Assert that the createWord() method works correctly for different word types
24          Noun noun = new Noun(wordString);
25          assertEquals(noun, wordFactory.createWord(wordString, WordType.NOUN));
26  
27          Adjective adjective = new Adjective(wordString);
28          assertEquals(adjective, wordFactory.createWord(wordString, WordType.ADJECTIVE));
29  
30          // Assert that the createWord() method works correctly for different tenses
31          for (Tense tense : Tense.values()) {
32              // Assert that the createWord() ignores the tense for Noun and Adjective
33              assertEquals(noun, wordFactory.createWord(wordString, WordType.NOUN, tense));
34              assertEquals(adjective, wordFactory.createWord(wordString, WordType.ADJECTIVE, tense));
35  
36              Verb verb = new Verb(wordString, tense);
37              assertEquals(verb, wordFactory.createWord(wordString, WordType.VERB, tense));
38          }
39          // Assert that the createWord() method throws an exception when creating a verb without a tense
40          assertThrows(IllegalArgumentException.class, () -> wordFactory.createWord(wordString, WordType.VERB));
41      }
